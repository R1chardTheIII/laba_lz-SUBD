SELECT LAST_NAME FROM STUD 
WHERE LAST_NAME REGEXP '[бо]';

SELECT * FROM STUD 
WHERE LEFT(LAST_NAME, 1) = 'К' AND S_NAME IS NULL;

SELECT * FROM STUD 
WHERE LENGTH(LAST_NAME) >= 8;

SELECT * FROM STUD 
WHERE LENGTH(LAST_NAME) NOT BETWEEN 7 AND 7 AND LOCATE('а', LAST_NAME) > 0;

SELECT * FROM STUD 
WHERE FACULTY = 'ФПМ' AND FORM = 'очная' AND (YEARR = 1 OR YEARR = 2)
ORDER BY S_NAME ASC;

SELECT * FROM STUD 
WHERE FACULTY = 'ФПК' AND FORM = 'заочная' AND EXM > 6.0
ORDER BY EXM DESC;

SELECT * FROM TEACH 
WHERE FACULTY = 'ФПК'
ORDER BY FORM ASC, LAST_NAME ASC;

SELECT * FROM TEACH 
WHERE FACULTY = 'ФПМ' AND YEARR = 1 AND HOURSS > 100;

SELECT * FROM TEACH 
WHERE S_NAME IS NULL AND START_WORK_DATE < (CURRENT_DATE() - INTERVAL 3 YEAR);

SELECT DISTINCT SUBJ FROM TEACH 
WHERE FACULTY = 'ФПМ' AND YEARR = 3;

SELECT YEARR, FORM, LAST_NAME, F_NAME, S_NAME, SUBJ, HOURSS FROM TEACH 
WHERE FACULTY = 'ФПК' AND HOURSS > 100;

SELECT FACULTY, YEARR, FORM, LAST_NAME, F_NAME, S_NAME, SUBJ FROM TEACH 
WHERE S_NAME IS NULL;

SELECT LAST_NAME, F_NAME, S_NAME FROM TEACH
WHERE (YEAR(CURRENT_DATE()) - YEAR(BR_DATE)) > 30;

SELECT LAST_NAME, F_NAME, S_NAME FROM TEACH 
WHERE TIMESTAMPDIFF(YEAR, BR_DATE, CURRENT_DATE()) BETWEEN 35 AND 40
ORDER BY LAST_NAME;

SELECT LAST_NAME, F_NAME, S_NAME FROM TEACH 
WHERE EXTRACT(MONTH FROM BR_DATE) = 10
ORDER BY BR_DATE;

SELECT FACULTY, AVG(EXM) AS FACULTY_AVG_SCORE FROM STUD
WHERE FORM = 'заочная' GROUP BY FACULTY;

SELECT FACULTY, YEARR, MAX(EXM) AS HIGHEST_SCORE FROM STUD
GROUP BY FACULTY, YEARR;

SELECT FACULTY, AVG(EXM) AS TOTAL_AVG FROM STUD
GROUP BY FACULTY HAVING TOTAL_AVG > 7;

SELECT YEARR, FACULTY, FORM, AVG(EXM) AS SCORE_AVG FROM STUD
GROUP BY YEARR, FACULTY, FORM HAVING AVG(EXM) > 7.5;

SELECT FACULTY, YEARR, MIN(EXM) AS LOWEST_EXM FROM STUD
GROUP BY FACULTY, YEARR;

SELECT FACULTY, FORM, MIN(EXM) AS FLOOR_EXM FROM STUD
GROUP BY FACULTY, FORM HAVING MIN(EXM) > 6;

SELECT AVG(ALL_H - INCLASS_H) AS SELF_STUDY_HOURS FROM STUD
WHERE FACULTY = 'ФПК' AND YEARR = 3 AND FORM = 'заочная';

SELECT FACULTY, YEARR, FORM, AVG(ALL_H - INCLASS_H) AS HOMEWORK_HOURS FROM STUD
GROUP BY FACULTY, YEARR, FORM HAVING HOMEWORK_HOURS > 150;

SELECT LAST_NAME, F_NAME, S_NAME, COUNT(DISTINCT SUBJ) AS UNIQUE_SUBJECTS FROM TEACH
GROUP BY LAST_NAME, F_NAME, S_NAME;

SELECT FACULTY, COUNT(DISTINCT LAST_NAME, F_NAME) AS TOTAL_TEACHERS FROM TEACH
GROUP BY FACULTY;

SELECT SUBJ, MAX(HOURSS) AS MAX_HOURS_LIMIT FROM TEACH
GROUP BY SUBJ;

SELECT LAST_NAME, F_NAME, S_NAME FROM TEACH
GROUP BY LAST_NAME, F_NAME, S_NAME
HAVING COUNT(DISTINCT SUBJ) >= 2;

SELECT FACULTY, YEARR, SUM(HOURSS) AS TOTAL_FACULTY_HOURS FROM TEACH
GROUP BY FACULTY, YEARR;

SELECT FACULTY, YEARR, COUNT(DISTINCT SUBJ) AS TOTAL_SUBJ FROM TEACH
WHERE YEARR = 2
GROUP BY FACULTY, YEARR ORDER BY FACULTY DESC, YEARR ASC;

SELECT FACULTY, COUNT(DISTINCT SUBJ) AS FOREIGN_TEACHER_SUBJ FROM TEACH
WHERE S_NAME IS NULL
GROUP BY FACULTY;
