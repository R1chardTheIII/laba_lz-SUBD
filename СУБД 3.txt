CREATE TABLE STUD (
	STUDENT_ID INT IDENTITY (1,1) NOT NULL PRIMARY KEY ,
	LAST_NAME NVARCHAR (30),
	F_NAME NVARCHAR (30),
	S_NAME NVARCHAR (30),
	BR_DATE DATE,
	IN_DATE DATE,
	EXM INT
);
CREATE TABLE FACULTY (
	FACULTYID INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
	FACULTY_NAME NVARCHAR(40)
);
CREATE TABLE FORM (
	FORMID INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
	FORM_NAME NVARCHAR (10)
);
CREATE TABLE HOURSS (
	HOURS_ID INT IDENTITY (1,1) NOT NULL PRIMARY KEY ,
	COURSE INT,
	FACULTY_ID INT,
	FORM_ID INT,
	ALL_H INT,
	INCLASS_H INT,
	FOREIGN KEY (FACULTY_ID) REFERENCES FACULTY(FACULTYID),
	FOREIGN KEY (FORM_ID) REFERENCES FORM(FORMID)
);
CREATE TABLE PROCESS (
	STUDENTID INT,
	HOURSID INT,
	PRIMARY KEY (STUDENTID,HOURSID),
	FOREIGN KEY (HOURSID) REFERENCES HOURSS(HOURS_ID),
	FOREIGN KEY (STUDENTID) REFERENCES STUD (STUDENT_ID)
	);

SELECT FACULTY_NAME, COURSE 
FROM FORM
JOIN HOURSS ON FORMID = FORM_ID
JOIN PROCESS ON HOURS_ID = HOURSID
JOIN STUD ON STUDENTID = STUDENT_ID
JOIN FACULTY ON FACULTY_ID = FACULTYID
WHERE FORM_NAME = 'Заочная' AND TIMESTAMPDIFF(YEAR, BR_DATE, CURRENT_DATE()) < 37;

SELECT FACULTY_NAME, COUNT(PROCESS.STUDENTID) AS STUDCOUNT 
FROM PROCESS
JOIN HOURSS ON HOURS_ID = HOURS_ID
RIGHT JOIN FACULTY ON FACULTY_ID = FACULTYID 
GROUP BY FACULTY_NAME;

SELECT FORM_NAME, COUNT(PROCESS.STUDENTID) AS STUDCOUNT 
FROM PROCESS
RIGHT JOIN HOURSS ON HOURS_ID = HOURSID
RIGHT JOIN FORM ON FORM_ID = FORMID 
GROUP BY FORM_NAME;

SELECT f.FACULTY_NAME, AVG(YEAR(CURRENT_DATE()) - YEAR(s.BR_DATE)) AS AVERAGE_AGE 
FROM FACULTY f, HOURSS h, PROCESS p, STUD s
WHERE f.FACULTYID = h.FACULTY_ID 
  AND h.HOURS_ID = p.HOURSID 
  AND p.STUDENTID = s.STUDENT_ID
GROUP BY f.FACULTY_NAME;

SELECT s.IN_DATE, f.FACULTY_NAME, h.COURSE, fr.FORM_NAME 
FROM HOURSS h
JOIN PROCESS p ON h.HOURS_ID = p.HOURSID
JOIN STUD s ON p.STUDENTID = s.STUDENT_ID
JOIN FACULTY f ON h.FACULTY_ID = f.FACULTYID
JOIN FORM fr ON h.FORM_ID = fr.FORMID
WHERE s.S_NAME IS NULL;

SELECT FACULTY_NAME, COUNT(STUDENT_ID) AS MAX_STUD_2015 
FROM STUD
JOIN PROCESS ON STUDENT_ID = STUDENTID
JOIN HOURSS ON HOURSID = HOURS_ID
JOIN FACULTY ON FACULTY_ID = FACULTYID
WHERE YEAR(IN_DATE) = 2015 
GROUP BY FACULTY_NAME  
HAVING COUNT(STUDENT_ID) = (
    SELECT MAX(STUDCOUNT) 
    FROM (
        SELECT COUNT(st.STUDENT_ID) AS STUDCOUNT 
        FROM STUD st
        JOIN PROCESS pr ON st.STUDENT_ID = pr.STUDENTID
        JOIN HOURSS hr ON pr.HOURSID = hr.HOURS_ID
        WHERE YEAR(st.IN_DATE) = 2015 
        GROUP BY hr.FACULTY_ID
    ) AS DOP
);


SELECT FACULTY_NAME, FORM_NAME, COUNT(STUDENTID) AS STUDCOUNT 
FROM HOURSS
JOIN FACULTY ON FACULTY_ID = FACULTYID
JOIN FORM ON FORM_ID = FORMID
LEFT JOIN PROCESS ON HOURS_ID = HOURSID
LEFT JOIN STUD ON STUDENTID = STUDENT_ID AND YEAR(IN_DATE) = 2014
GROUP BY FACULTY_NAME, FORM_NAME;

SELECT FACULTY_NAME FROM FACULTY 
WHERE FACULTYID IN (
    SELECT DISTINCT FACULTY_ID FROM HOURSS 
    JOIN FORM ON FORM_ID = FORMID 
    WHERE FORM_NAME = 'Заочная'
);

SELECT f.FACULTY_NAME, fr.FORM_NAME, h.COURSE 
FROM FACULTY f, HOURSS h, FORM fr
WHERE f.FACULTYID = h.FACULTY_ID AND h.FORM_ID = fr.FORMID;

SELECT FACULTY_NAME, FORM_NAME, COUNT(STUDENTID) AS STUDCOUNT 
FROM PROCESS
RIGHT JOIN HOURSS ON HOURS_ID = HOURSID
RIGHT JOIN FACULTY ON FACULTY_ID = FACULTYID
RIGHT JOIN FORM ON FORM_ID = FORMID
GROUP BY FACULTY_NAME, FORM_NAME;

SELECT FACULTY_NAME, FORM_NAME, COURSE, COUNT(STUDENTID) AS STUDCOUNT 
FROM PROCESS
RIGHT JOIN HOURSS ON HOURS_ID = HOURSID
JOIN FACULTY ON FACULTY_ID = FACULTYID
JOIN FORM ON FORM_ID = FORMID
WHERE COURSE = 1 OR COURSE = 3 
GROUP BY FACULTY_NAME, FORM_NAME, COURSE;

SELECT FACULTY_NAME, COURSE, COUNT(s.STUDENT_ID) AS STUDCOUNT 
FROM HOURSS h
JOIN FACULTY f ON h.FACULTY_ID = f.FACULTYID
LEFT JOIN PROCESS p ON h.HOURS_ID = p.HOURSID
LEFT JOIN STUD s ON p.STUDENTID = s.STUDENT_ID AND s.S_NAME IS NULL
GROUP BY FACULTY_NAME, COURSE;

SELECT h.COURSE, f.FACULTY_NAME, COUNT(s.STUDENT_ID) AS STUDCOUNT 
FROM STUD s
JOIN PROCESS p ON s.STUDENT_ID = p.STUDENTID AND s.EXM > 7.5
RIGHT JOIN HOURSS h ON p.HOURSID = h.HOURS_ID
JOIN FACULTY f ON h.FACULTY_ID = f.FACULTYID
GROUP BY f.FACULTY_NAME, h.COURSE;

SELECT FACULTY_NAME, FORM_NAME, COUNT(STUDENTID) AS OLD_STUDENT 
FROM PROCESS
LEFT JOIN STUD ON STUDENTID = STUDENT_ID AND TIMESTAMPDIFF(YEAR, BR_DATE, CURRENT_DATE()) > 45
RIGHT JOIN HOURSS ON HOURS_ID = HOURSID
JOIN FACULTY ON FACULTY_ID = FACULTYID
JOIN FORM ON FORM_ID = FORMID
GROUP BY FACULTY_NAME, FORM_NAME;

SELECT f.FACULTY_NAME, fr.FORM_NAME, h.COURSE, COUNT(s.STUDENT_ID) AS YOUNG_STUDENT 
FROM STUD s 
JOIN PROCESS p ON s.STUDENT_ID = p.STUDENTID AND TIMESTAMPDIFF(YEAR, s.BR_DATE, CURRENT_DATE()) < 27
RIGHT JOIN HOURSS h ON p.HOURSID = h.HOURS_ID
JOIN FACULTY f ON h.FACULTY_ID = f.FACULTYID
JOIN FORM fr ON h.FORM_ID = fr.FORMID
GROUP BY f.FACULTY_NAME, fr.FORM_NAME, h.COURSE;

SELECT FACULTY_NAME, COUNT(STUDENTID) AS STUDCOUNT 
FROM PROCESS
JOIN STUD ON STUDENTID = STUDENT_ID AND LEFT(LAST_NAME, 1) = 'С'
RIGHT JOIN HOURSS ON HOURS_ID = HOURSID
RIGHT JOIN FACULTY ON FACULTY_ID = FACULTYID
GROUP BY FACULTY_NAME;
