CREATE TABLE STUD (
	STUDENT_ID INT IDENTITY (1,1) NOT NULL PRIMARY KEY ,
	LAST_NAME NVARCHAR (30),
	F_NAME NVARCHAR (30),
	S_NAME NVARCHAR (30),
	BR_DATE DATE,
	IN_DATE DATE,
	EXM INT
);


CREATE TABLE FACULTY (
	FACULTYID INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
	FACULTY_NAME NVARCHAR(35)
);

CREATE TABLE FORM (
	FORMID INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
	FORM_NAME NVARCHAR (9)
);

CREATE TABLE HOURSS (
	HOURS_ID INT IDENTITY (1,1) NOT NULL PRIMARY KEY ,
	COURSE INT,
	FACULTY_ID INT,
	FORM_ID INT,
	ALL_H INT,
	INCLASS_H INT,
	FOREIGN KEY (FACULTY_ID) REFERENCES FACULTY(FACULTYID),
	FOREIGN KEY (FORM_ID) REFERENCES FORM(FORMID)
);

CREATE TABLE PROCESS (
	STUDENTID INT,
	HOURSID INT,
	PRIMARY KEY (STUDENTID,HOURSID),
	FOREIGN KEY (HOURSID) REFERENCES HOURSS(HOURS_ID),
	FOREIGN KEY (STUDENTID) REFERENCES STUD (STUDENT_ID)
	);

--хранимая процедура для подсчета студентов на факультете и форме обучения 
GO
CREATE PROCEDURE STUDCOUNT
    @FACULTYNAME NVARCHAR(25),
    @FORMNAME NVARCHAR(7)
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @COUNT INT = 0;

    SELECT @COUNT = COUNT(DISTINCT STUDENTID) 
    FROM PROCESS 
    JOIN HOURSS ON HOURSID = HOURS_ID
    JOIN FACULTY ON FACULTY_ID = FACULTYID
    JOIN FORM ON FORM_ID = FORMID 
    WHERE FACULTY_NAME = @FACULTYNAME AND FORM_NAME = @FORMNAME;

    SELECT @COUNT AS STUDENT_COUNT;
END
GO
EXECUTE STUDCOUNT 'ФПИ','Заочная';

--хранимая процедура для подсчета числа предметов на факультетах, а также числа одинаковых предметов
GO
CREATE PROCEDURE SUBCOUNT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @FPK INT = 0;
    DECLARE @FPI INT = 0;
    DECLARE @ALL INT = 0;
    DECLARE @SAME INT = 0;

    SELECT @FPK = COUNT(*) FROM HOURSS WHERE FACULTY_ID = 1;
    SELECT @FPI = COUNT(*) FROM HOURSS WHERE FACULTY_ID = 2;
    SELECT @ALL = COUNT(*) FROM HOURSS;
    
    SELECT @SAME = COUNT(*) 
    FROM HOURSS h1
    JOIN HOURSS h2 ON h1.COURSE = h2.COURSE AND h1.FORM_ID = h2.FORM_ID
    WHERE h1.FACULTY_ID < h2.FACULTY_ID;

    -- Использование CONCAT предотвращает появление лишних пробелов, в отличие от STR()
    PRINT CONCAT('Для ФПК читается предметов: ', @FPK);
    PRINT CONCAT('Для ФПИ читается предметов: ', @FPI);
    PRINT CONCAT('Всего предметов в базе: ', @ALL);
    PRINT CONCAT('Из них идентичных: ', @SAME);
END
GO

--добавление студента на указанный факультет и форму обучения
GO
CREATE PROCEDURE NEWSTUDENT
    @FACULTYNAME NVARCHAR(35),
    @FORMNAME NVARCHAR(9),
    @BRDATE DATE,
    @INDATE DATE,
    @LASTNAME NVARCHAR(30),
    @FNAME NVARCHAR(30),
    @SNAME NVARCHAR(30) 
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @FACULTYID INT, @FORMID INT, @HOURSID INT, @NEWID INT;

    -- Проверка существования факультета
    SELECT @FACULTYID = FACULTYID FROM FACULTY WHERE FACULTY_NAME = @FACULTYNAME;
    IF @FACULTYID IS NULL
    BEGIN
        PRINT 'Ошибка: Факультет не найден.';
        RETURN;
    END

    -- Проверка существования формы обучения
    SELECT @FORMID = FORMID FROM FORM WHERE FORM_NAME = @FORMNAME;
    IF @FORMID IS NULL
    BEGIN
        PRINT 'Ошибка: Форма обучения не найдена.';
        RETURN;
    END

    -- Проверка существования учебного плана (курса) для 1-го курса
    SELECT @HOURSID = HOURS_ID FROM HOURSS 
    WHERE FACULTY_ID = @FACULTYID AND FORM_ID = @FORMID AND COURSE = 1;
    IF @HOURSID IS NULL
    BEGIN
        PRINT 'Ошибка: Учебный план для 1-го курса не найден.';
        RETURN;
    END

    -- Вставка нового студента
    INSERT INTO STUD (LAST_NAME, F_NAME, S_NAME, BR_DATE, IN_DATE, EXM)
    VALUES (@LASTNAME, @FNAME, @SNAME, @BRDATE, @INDATE, 0);

    -- Безопасное получение ID только что добавленного студента в рамках текущей сессии
    SET @NEWID = SCOPE_IDENTITY();

    -- Привязка студента к процессу обучения
    INSERT INTO PROCESS (STUDENTID, HOURSID)
    VALUES (@NEWID, @HOURSID);

    PRINT 'Студент успешно добавлен.';
END
GO

@LASTNAME='Мазур', @FNAME='Даниил', @SNAME='Богданович';
