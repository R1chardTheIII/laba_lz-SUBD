CREATE TABLE Employees (
Employee_id INT IDENTITY(1,1) PRIMARY KEY,
F_Name NVARCHAR(20) NOT NULL,
L_Name NVARCHAR(20) NOT NULL,
BDate DATE NOT NULL
);

CREATE TABLE Categories (
Categ_id INT IDENTITY(1,1) PRIMARY KEY,
Categ_Name NVARCHAR(20) NOT NULL);

CREATE TABLE Products (
Product_id INT IDENTITY(1,1) PRIMARY KEY, 
Product_Name NVARCHAR(20) NOT NULL,
Categ_id INT NOT NULL,
Price Money NOT NULL,
CONSTRAINT fk_Products_Categories FOREIGN KEY (Categ_id) 
REFERENCES Categories(Categ_id));

CREATE FUNCTION Age_emp (@BDate DATE)
RETURNS INT
AS
BEGIN
    DECLARE @Age INT;
    SET @Age = DATEDIFF(YEAR, @BDAte, GETDATE());
    RETURN @Age
END;
GO
SELECT F_Name, L_Name, dbo.Age_emp(BDate) FROM Employees WHERE dbo.Age_emp(BDate) > 30

CREATE FUNCTION Choose_Categ (@Categ_Name NVARCHAR(20))
RETURNS TABLE
AS
RETURN
    (SELECT P.Product_id, P.Product_Name, P.Price FROM Products P
    INNER JOIN Categories C ON P.Categ_id = C.Categ_id
    WHERE C.Categ_Name = @Categ_Name);
GO
SELECT Product_id, Product_Name, Price FROM Choose_Categ('Телефоны');


Go
CREATE FUNCTION Super_Name (@F_Name NVARCHAR(20), @L_Name NVARCHAR(20))
RETURNS NVARCHAR(40)
AS
BEGIN
	RETURN @F_Name + ' ' + @L_Name
END;
GO
SELECT F_Name, L_Name, dbo.Super_Name(F_Name, L_Name) AS Full_Name FROM Employees;

 
Go
CREATE FUNCTION AvrPrice (@Categ_id INT)
RETURNS Money
AS
BEGIN
    RETURN (SELECT ISNULL(AVG(Price), 0) 
            FROM Products 
            WHERE Categ_id = @Categ_id);
END;
GO
SELECT Categ_Name, dbo.AvrPrice(Categ_id) AS Avr_Price FROM Categories;
